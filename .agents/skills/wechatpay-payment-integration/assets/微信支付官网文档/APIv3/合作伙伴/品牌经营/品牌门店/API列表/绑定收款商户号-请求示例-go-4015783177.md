func BindRecipient(config *wxpay_utility.MchConfig, request *BindRecipientRequest) (response *BrandStoresBindRecipientResponse, err error) {
	const (
		host   = "https://api.mch.weixin.qq.com"
		method = "POST"
		path   = "/v3/brand/partner/store/brandstores/{store_id}/bindrecipient"
	)

	reqUrl, err := url.Parse(fmt.Sprintf("%s%s", host, path))
	if err != nil {
		return nil, err
	}
	reqUrl.Path = strings.Replace(reqUrl.Path, "{store_id}", url.PathEscape(*request.StoreId), -1)
	reqBody, err := json.Marshal(request)
	if err != nil {
		return nil, err
	}
	httpRequest, err := http.NewRequest(method, reqUrl.String(), bytes.NewReader(reqBody))
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Accept", "application/json")
	httpRequest.Header.Set("Wechatpay-Serial", config.WechatPayPublicKeyId())
	httpRequest.Header.Set("Content-Type", "application/json")
	authorization, err := wxpay_utility.BuildAuthorization(config.MchId(), config.CertificateSerialNo(), config.PrivateKey(), method, reqUrl.RequestURI(), reqBody)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Authorization", authorization)

	client := &http.Client{}
	httpResponse, err := client.Do(httpRequest)
	if err != nil {
		return nil, err
	}
	respBody, err := wxpay_utility.ExtractResponseBody(httpResponse)
	if err != nil {
		return nil, err
	}
	if httpResponse.StatusCode >= 200 && httpResponse.StatusCode < 300 {
		// 2XX 成功，验证应答签名
		err = wxpay_utility.ValidateResponse(
			config.WechatPayPublicKeyId(),
			config.WechatPayPublicKey(),
			&httpResponse.Header,
			respBody,
		)
		if err != nil {
			return nil, err
		}
		response := &BrandStoresBindRecipientResponse{}
		if err := json.Unmarshal(respBody, response); err != nil {
			return nil, err
		}

		return response, nil
	} else {
		return nil, wxpay_utility.NewApiException(
			httpResponse.StatusCode,
			httpResponse.Header,
			respBody,
		)
	}
}

type BindRecipientRequest struct {
	BrandId     *string `json:"brand_id,omitempty"`
	StoreId     *string `json:"store_id,omitempty"`
	Mchid       *string `json:"mchid,omitempty"`
	CompanyName *string `json:"company_name,omitempty"`
}

func (o *BindRecipientRequest) MarshalJSON() ([]byte, error) {
	type Alias BindRecipientRequest
	a := &struct {
		StoreId *string `json:"store_id,omitempty"`
		*Alias
	}{
		// 序列化时移除非 Body 字段
		StoreId: nil,
		Alias:   (*Alias)(o),
	}
	return json.Marshal(a)
}

type BrandStoresBindRecipientResponse struct {
	BrandId        *string         `json:"brand_id,omitempty"`
	StoreId        *string         `json:"store_id,omitempty"`
	Mchid          *string         `json:"mchid,omitempty"`
	CompanyName    *string         `json:"company_name,omitempty"`
	RecipientState *RecipientState `json:"recipient_state,omitempty"`
}

type RecipientState string

func (e RecipientState) Ptr() *RecipientState {
	return &e
}

const (
	RECIPIENTSTATE_CONFIRMED        RecipientState = "CONFIRMED"
	RECIPIENTSTATE_ADMIN_REJECTED   RecipientState = "ADMIN_REJECTED"
	RECIPIENTSTATE_CONFIRMING       RecipientState = "CONFIRMING"
	RECIPIENTSTATE_TIMEOUT_REJECTED RecipientState = "TIMEOUT_REJECTED"
)
