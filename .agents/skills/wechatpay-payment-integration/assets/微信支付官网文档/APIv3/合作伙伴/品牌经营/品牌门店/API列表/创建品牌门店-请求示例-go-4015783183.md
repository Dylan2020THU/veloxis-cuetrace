func CreateBrandStore(config *wxpay_utility.MchConfig, request *BrandStoresCreateRequest) (response *BrandStoresEntity, err error) {
	const (
		host   = "https://api.mch.weixin.qq.com"
		method = "POST"
		path   = "/v3/brand/partner/store/brandstores"
	)

	reqUrl, err := url.Parse(fmt.Sprintf("%s%s", host, path))
	if err != nil {
		return nil, err
	}
	reqBody, err := json.Marshal(request)
	if err != nil {
		return nil, err
	}
	httpRequest, err := http.NewRequest(method, reqUrl.String(), bytes.NewReader(reqBody))
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Accept", "application/json")
	httpRequest.Header.Set("Wechatpay-Serial", config.WechatPayPublicKeyId())
	httpRequest.Header.Set("Content-Type", "application/json")
	authorization, err := wxpay_utility.BuildAuthorization(config.MchId(), config.CertificateSerialNo(), config.PrivateKey(), method, reqUrl.RequestURI(), reqBody)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Authorization", authorization)

	client := &http.Client{}
	httpResponse, err := client.Do(httpRequest)
	if err != nil {
		return nil, err
	}
	respBody, err := wxpay_utility.ExtractResponseBody(httpResponse)
	if err != nil {
		return nil, err
	}
	if httpResponse.StatusCode >= 200 && httpResponse.StatusCode < 300 {
		// 2XX 成功，验证应答签名
		err = wxpay_utility.ValidateResponse(
			config.WechatPayPublicKeyId(),
			config.WechatPayPublicKey(),
			&httpResponse.Header,
			respBody,
		)
		if err != nil {
			return nil, err
		}
		response := &BrandStoresEntity{}
		if err := json.Unmarshal(respBody, response); err != nil {
			return nil, err
		}

		return response, nil
	} else {
		return nil, wxpay_utility.NewApiException(
			httpResponse.StatusCode,
			httpResponse.Header,
			respBody,
		)
	}
}

type BrandStoresCreateRequest struct {
	BrandId       *string        `json:"brand_id,omitempty"`
	StoreBasics   *StoreBase     `json:"store_basics,omitempty"`
	StoreAddress  *StoreLocation `json:"store_address,omitempty"`
	StoreBusiness *StoreBusiness `json:"store_business,omitempty"`
}

type BrandStoresEntity struct {
	BrandId            *string          `json:"brand_id,omitempty"`
	StoreId            *string          `json:"store_id,omitempty"`
	StoreState         *StoreState      `json:"store_state,omitempty"`
	AuditState         *AuditState      `json:"audit_state,omitempty"`
	ReviewRejectReason *string          `json:"review_reject_reason,omitempty"`
	StoreBasics        *StoreBase       `json:"store_basics,omitempty"`
	StoreAddress       *StoreLocation   `json:"store_address,omitempty"`
	StoreBusiness      *StoreBusiness   `json:"store_business,omitempty"`
	StoreRecipient     []StoreRecipient `json:"store_recipient,omitempty"`
}

type StoreBase struct {
	StoreReferenceId *string `json:"store_reference_id,omitempty"`
	BranchName       *string `json:"branch_name,omitempty"`
}

type StoreLocation struct {
	AddressCode        *string `json:"address_code,omitempty"`
	AddressDetail      *string `json:"address_detail,omitempty"`
	AddressComplements *string `json:"address_complements,omitempty"`
	Longitude          *string `json:"longitude,omitempty"`
	Latitude           *string `json:"latitude,omitempty"`
}

type StoreBusiness struct {
	ServicePhone  *string `json:"service_phone,omitempty"`
	BusinessHours *string `json:"business_hours,omitempty"`
}

type StoreState string

func (e StoreState) Ptr() *StoreState {
	return &e
}

const (
	STORESTATE_OPEN     StoreState = "OPEN"
	STORESTATE_CREATING StoreState = "CREATING"
	STORESTATE_CLOSED   StoreState = "CLOSED"
)

type AuditState string

func (e AuditState) Ptr() *AuditState {
	return &e
}

const (
	AUDITSTATE_SUCCESS    AuditState = "SUCCESS"
	AUDITSTATE_PROCESSING AuditState = "PROCESSING"
	AUDITSTATE_REJECTED   AuditState = "REJECTED"
)

type StoreRecipient struct {
	Mchid          *string         `json:"mchid,omitempty"`
	CompanyName    *string         `json:"company_name,omitempty"`
	RecipientState *RecipientState `json:"recipient_state,omitempty"`
}

type RecipientState string

func (e RecipientState) Ptr() *RecipientState {
	return &e
}

const (
	RECIPIENTSTATE_CONFIRMED        RecipientState = "CONFIRMED"
	RECIPIENTSTATE_ADMIN_REJECTED   RecipientState = "ADMIN_REJECTED"
	RECIPIENTSTATE_CONFIRMING       RecipientState = "CONFIRMING"
	RECIPIENTSTATE_TIMEOUT_REJECTED RecipientState = "TIMEOUT_REJECTED"
)
