  -H "Accept: application/json" \
  -H "Content-Type: application/json" \
  -d '{
    "product_coupon_id" : "1002323",
    "stock_bundle_id" : "100232301",
    "appid" : "wx233544546545989",
    "out_request_no" : "1002600620019090123144054436",
