## curl


> POST

发送会员券

```
curl -X POST \
  https://api.mch.weixin.qq.com/v3/marketing/partner/product-coupon/users/oh-394z-6CGkNoJrsDLTTUKiAnp4/coupons \
  -H "Authorization: WECHATPAY2-SHA256-RSA2048 mchid=\"1900000001\",..." \
  -H "Accept: application/json" \
  -H "Content-Type: application/json" \
  -d '{
    "coupon_code" : "Code_123456",
    "product_coupon_id" : "1000000013",
    "appid" : "wx233544546545989",
    "attach" : "any attach content",
    "coupon_tag_info" : {
      "coupon_tag_list" : [
        "MEMBER"
      ],
      "member_tag_info" : {
        "member_card_id" : "MemberCardId_1234567890"
      }
    },
    "stock_id" : "1000000013001",
    "send_request_no" : "MCHSEND202003101234",
    "brand_id" : "120344"
  }'

```
