## Java


需配合微信支付工具库 WXPayUtility 使用，请参考 [Java](https://pay.weixin.qq.com/doc/v3/partner/4014985777.md)

package com.java.demo;

import com.java.utils.WXPayUtility; // 引用微信支付工具库，参考：https://pay.weixin.qq.com/doc/v3/partner/4014985777

import com.google.gson.annotations.SerializedName;
import com.google.gson.annotations.Expose;
import okhttp3.MediaType;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

import java.io.IOException;
import java.io.UncheckedIOException;
import java.security.PrivateKey;
import java.security.PublicKey;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * 查询品牌门店列表
 */
public class ListBrandStores {
  private static String HOST = "https://api.mch.weixin.qq.com";
  private static String METHOD = "GET";
  private static String PATH = "/v3/brand/partner/store/brandstores";

  public static void main(String[] args) {
    // TODO: 请准备商户开发必要参数，参考：https://pay.weixin.qq.com/doc/v3/partner/4013080340
    ListBrandStores client = new ListBrandStores(
      "19xxxxxxxx",                    // 商户号，是由微信支付系统生成并分配给每个商户的唯一标识符，商户号获取方式参考 https://pay.weixin.qq.com/doc/v3/partner/4013080340
      "1DDE55AD98Exxxxxxxxxx",         // 商户API证书序列号，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4013058924
      "/path/to/apiclient_key.pem",     // 商户API证书私钥文件路径，本地文件路径
      "PUB_KEY_ID_xxxxxxxxxxxxx",      // 微信支付公钥ID，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4013038589
      "/path/to/wxp_pub.pem"           // 微信支付公钥文件路径，本地文件路径
    );

    ListBrandStoresRequest request = new ListBrandStoresRequest();
    request.brandId = "123456789";
    request.storeState = StoreState.OPEN;
    request.offset = 100L;
    request.limit = 20L;
    try {
      BrandStoresListResponse response = client.run(request);
        // TODO: 请求成功，继续业务逻辑
        System.out.println(response);
    } catch (WXPayUtility.ApiException e) {
        // TODO: 请求失败，根据状态码执行不同的逻辑
        e.printStackTrace();
    }
  }

  public BrandStoresListResponse run(ListBrandStoresRequest request) {
    String uri = PATH;
    Map<String, Object> args = new HashMap<>();
    args.put("brand_id", request.brandId);
    args.put("store_state", request.storeState);
    args.put("offset", request.offset);
    args.put("limit", request.limit);
    String queryString = WXPayUtility.urlEncode(args);
    if (!queryString.isEmpty()) {
        uri = uri + "?" + queryString;
    }

    Request.Builder reqBuilder = new Request.Builder().url(HOST + uri);
    reqBuilder.addHeader("Accept", "application/json");
    reqBuilder.addHeader("Wechatpay-Serial", wechatPayPublicKeyId);
    reqBuilder.addHeader("Authorization", WXPayUtility.buildAuthorization(mchid, certificateSerialNo, privateKey, METHOD, uri, null));
    reqBuilder.method(METHOD, null);
    Request httpRequest = reqBuilder.build();

    // 发送HTTP请求
    OkHttpClient client = new OkHttpClient.Builder().build();
    try (Response httpResponse = client.newCall(httpRequest).execute()) {
      String respBody = WXPayUtility.extractBody(httpResponse);
      if (httpResponse.code() >= 200 && httpResponse.code() < 300) {
        // 2XX 成功，验证应答签名
        WXPayUtility.validateResponse(this.wechatPayPublicKeyId, this.wechatPayPublicKey,
            httpResponse.headers(), respBody);

        // 从HTTP应答报文构建返回数据
        return WXPayUtility.fromJson(respBody, BrandStoresListResponse.class);
      } else {
        throw new WXPayUtility.ApiException(httpResponse.code(), respBody, httpResponse.headers());
      }
    } catch (IOException e) {
      throw new UncheckedIOException("Sending request to " + uri + " failed.", e);
    }
  }

  private final String mchid;
  private final String certificateSerialNo;
  private final PrivateKey privateKey;
  private final String wechatPayPublicKeyId;
  private final PublicKey wechatPayPublicKey;

  public ListBrandStores(String mchid, String certificateSerialNo, String privateKeyFilePath, String wechatPayPublicKeyId, String wechatPayPublicKeyFilePath) {
    this.mchid = mchid;
    this.certificateSerialNo = certificateSerialNo;
    this.privateKey = WXPayUtility.loadPrivateKeyFromPath(privateKeyFilePath);
    this.wechatPayPublicKeyId = wechatPayPublicKeyId;
    this.wechatPayPublicKey = WXPayUtility.loadPublicKeyFromPath(wechatPayPublicKeyFilePath);
  }

  public static class ListBrandStoresRequest {
    @SerializedName("brand_id")
    @Expose(serialize = false)
    public String brandId;

    @SerializedName("offset")
    @Expose(serialize = false)
    public Long offset;

    @SerializedName("limit")
    @Expose(serialize = false)
    public Long limit;

    @SerializedName("store_state")
    @Expose(serialize = false)
    public StoreState storeState;
  }

  public static class BrandStoresListResponse {
    @SerializedName("brand_id")
    public String brandId;

    @SerializedName("data")
    public List<BrandStoresEntity> data;

    @SerializedName("offset")
    public Long offset;

    @SerializedName("limit")
    public Long limit;

    @SerializedName("total_count")
    public Long totalCount;
  }

  public enum StoreState {
    @SerializedName("OPEN")
    OPEN,
    @SerializedName("CREATING")
    CREATING,
    @SerializedName("CLOSED")
    CLOSED
  }

  public static class BrandStoresEntity {
    @SerializedName("brand_id")
    public String brandId;

    @SerializedName("store_id")
    public String storeId;

    @SerializedName("store_state")
    public StoreState storeState;

    @SerializedName("audit_state")
    public AuditState auditState;

    @SerializedName("review_reject_reason")
    public String reviewRejectReason;

    @SerializedName("store_basics")
    public StoreBase storeBasics;

    @SerializedName("store_address")
    public StoreLocation storeAddress;

    @SerializedName("store_business")
    public StoreBusiness storeBusiness;

    @SerializedName("store_recipient")
    public List<StoreRecipient> storeRecipient;
  }

  public enum AuditState {
    @SerializedName("SUCCESS")
    SUCCESS,
    @SerializedName("PROCESSING")
    PROCESSING,
    @SerializedName("REJECTED")
    REJECTED
  }

  public static class StoreBase {
    @SerializedName("store_reference_id")
    public String storeReferenceId;

    @SerializedName("branch_name")
    public String branchName;
  }

  public static class StoreLocation {
    @SerializedName("address_code")
    public String addressCode;

    @SerializedName("address_detail")
    public String addressDetail;

    @SerializedName("address_complements")
    public String addressComplements;

    @SerializedName("longitude")
    public String longitude;

    @SerializedName("latitude")
    public String latitude;
  }

  public static class StoreBusiness {
    @SerializedName("service_phone")
    public String servicePhone;

    @SerializedName("business_hours")
    public String businessHours;
  }

  public static class StoreRecipient {
    @SerializedName("mchid")
    public String mchid;

    @SerializedName("company_name")
    public String companyName;

    @SerializedName("recipient_state")
    public RecipientState recipientState;
  }

  public enum RecipientState {
    @SerializedName("CONFIRMED")
    CONFIRMED,
    @SerializedName("ADMIN_REJECTED")
    ADMIN_REJECTED,
    @SerializedName("CONFIRMING")
    CONFIRMING,
    @SerializedName("TIMEOUT_REJECTED")
    TIMEOUT_REJECTED
  }

}
