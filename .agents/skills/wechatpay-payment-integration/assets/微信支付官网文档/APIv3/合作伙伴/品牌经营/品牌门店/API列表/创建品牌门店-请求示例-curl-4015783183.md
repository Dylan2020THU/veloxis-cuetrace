  -H "Accept: application/json" \
  -H "Content-Type: application/json" \
  -d '{
    "brand_id" : "123456789",
    "store_basics" : {
      "store_reference_id" : "MDL001",
      "branch_name" : "海岸城店"
    },

      "address_complements" : "地铁A口右侧100米",
