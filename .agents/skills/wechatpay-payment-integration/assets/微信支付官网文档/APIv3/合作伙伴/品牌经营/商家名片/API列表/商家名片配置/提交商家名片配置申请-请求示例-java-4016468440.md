## Java


需配合微信支付工具库 WXPayUtility 使用，请参考 [Java](https://pay.weixin.qq.com/doc/v3/partner/4014985777.md)

package com.java.demo;

import com.java.utils.WXPayUtility; // 引用微信支付工具库，参考：https://pay.weixin.qq.com/doc/v3/partner/4014985777

/**
 * 提交商家名片配置申请
 */
public class SubmitMchCardConfig {
  private static String HOST = "https://api.mch.weixin.qq.com";
  private static String METHOD = "POST";
  private static String PATH = "/v3/brand/card/card-configs";

  public static void main(String[] args) {
    // TODO: 请准备商户开发必要参数，参考：https://pay.weixin.qq.com/doc/v3/partner/4013080340
    SubmitMchCardConfig client = new SubmitMchCardConfig(
      "19xxxxxxxx",                    // 商户号，是由微信支付系统生成并分配给每个商户的唯一标识符，商户号获取方式参考 https://pay.weixin.qq.com/doc/v3/partner/4013080340
      "1DDE55AD98Exxxxxxxxxx",         // 商户API证书序列号，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4013058924
      "/path/to/apiclient_key.pem",     // 商户API证书私钥文件路径，本地文件路径
      "PUB_KEY_ID_xxxxxxxxxxxxx",      // 微信支付公钥ID，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4013038589
      "/path/to/wxp_pub.pem"           // 微信支付公钥文件路径，本地文件路径
    );

    SubmitMchCardConfigRequest request = new SubmitMchCardConfigRequest();
    request.businessCode = "1900013511_10000";
    request.brandId = "1234567";
    request.brandMiniProgramInfo = new MiniProgramInfo();
    request.brandMiniProgramInfo.appid = "wx1234567890abcdef";
    request.brandMiniProgramInfo.defaultJumpPath = "pages/shop/index";
    request.brandMiniProgramInfo.buttonText = "前往小程序";
    request.brandCustomerService = new MerchantCardCustomerService();
    request.brandCustomerService.customerServiceType = CustomerServiceType.MINI_PROGRAM;
    request.brandCustomerService.customerServicePhone = "12345678900";
    request.brandCustomerService.customerServicePath = "pages/service/index";
    request.brandCustomerService.appid = "wx1234567890abcdef";
    request.serviceList = new ArrayList<>();
    {
      MerchantCardService serviceListItem = new MerchantCardService();
      serviceListItem.serviceClassifyName = "会员服务";
      serviceListItem.serviceName = "会员中心";
      serviceListItem.serviceJumpType = ServiceJumpType.JUMP_MINI_PROGRAM;
      serviceListItem.serviceJumpPath = "pages/shop/index";
      serviceListItem.appid = "wx1234567890abcdef";

    try {
      SubmitMchCardConfigResponse response = client.run(request);
        // TODO: 请求成功，继续业务逻辑
        System.out.println(response);
    } catch (WXPayUtility.ApiException e) {
        // TODO: 请求失败，根据状态码执行不同的逻辑
        e.printStackTrace();
    }
  }

  public SubmitMchCardConfigResponse run(SubmitMchCardConfigRequest request) {
    String uri = PATH;
    String reqBody = WXPayUtility.toJson(request);

    Request.Builder reqBuilder = new Request.Builder().url(HOST + uri);
    reqBuilder.addHeader("Accept", "application/json");
    reqBuilder.addHeader("Wechatpay-Serial", wechatPayPublicKeyId);
    reqBuilder.addHeader("Authorization", WXPayUtility.buildAuthorization(mchid, certificateSerialNo,privateKey, METHOD, uri, reqBody));
    reqBuilder.addHeader("Content-Type", "application/json");
    RequestBody requestBody = RequestBody.create(MediaType.parse("application/json; charset=utf-8"), reqBody);
    reqBuilder.method(METHOD, requestBody);
    Request httpRequest = reqBuilder.build();

    // 发送HTTP请求
    OkHttpClient client = new OkHttpClient.Builder().build();
    try (Response httpResponse = client.newCall(httpRequest).execute()) {
      String respBody = WXPayUtility.extractBody(httpResponse);
      if (httpResponse.code() >= 200 && httpResponse.code() < 300) {
        // 2XX 成功，验证应答签名
        WXPayUtility.validateResponse(this.wechatPayPublicKeyId, this.wechatPayPublicKey,
            httpResponse.headers(), respBody);

        // 从HTTP应答报文构建返回数据
        return WXPayUtility.fromJson(respBody, SubmitMchCardConfigResponse.class);
      } else {
        throw new WXPayUtility.ApiException(httpResponse.code(), respBody, httpResponse.headers());
      }
    } catch (IOException e) {
      throw new UncheckedIOException("Sending request to " + uri + " failed.", e);
    }
  }

  private final String mchid;
  private final String certificateSerialNo;
  private final PrivateKey privateKey;
  private final String wechatPayPublicKeyId;
  private final PublicKey wechatPayPublicKey;

  public SubmitMchCardConfig(String mchid, String certificateSerialNo, String privateKeyFilePath, String wechatPayPublicKeyId, String wechatPayPublicKeyFilePath) {
    this.mchid = mchid;
    this.certificateSerialNo = certificateSerialNo;
    this.privateKey = WXPayUtility.loadPrivateKeyFromPath(privateKeyFilePath);
    this.wechatPayPublicKeyId = wechatPayPublicKeyId;
    this.wechatPayPublicKey = WXPayUtility.loadPublicKeyFromPath(wechatPayPublicKeyFilePath);
  }

  public static class SubmitMchCardConfigRequest {
    @SerializedName("business_code")
    public String businessCode;

    public String brandId;

    public MiniProgramInfo brandMiniProgramInfo;

    @SerializedName("brand_customer_service")

    @SerializedName("service_list")

  }

  public static class SubmitMchCardConfigResponse {

  }

    public String defaultJumpPath;

    @SerializedName("button_text")

  }

  public static class MerchantCardCustomerService {

    @SerializedName("customer_service_phone")
    public String customerServicePhone;

    @SerializedName("customer_service_path")
    public String customerServicePath;

    @SerializedName("appid")
    public String appid;
  }

  public static class MerchantCardService {
    @SerializedName("service_classify_name")
    public String serviceClassifyName;

    @SerializedName("service_name")
    public String serviceName;

    @SerializedName("service_jump_type")
    public ServiceJumpType serviceJumpType;

    @SerializedName("service_jump_path")
    public String serviceJumpPath;

    @SerializedName("appid")
    public String appid;
  }

  public enum CustomerServiceType {
    @SerializedName("MINI_PROGRAM")
    MINI_PROGRAM,
    @SerializedName("WECOM")
    WECOM,
    @SerializedName("CUSTOMIZE_WEB")
    CUSTOMIZE_WEB,
    @SerializedName("CUSTOMIZE_MP")
    CUSTOMIZE_MP,
    @SerializedName("SERVICE_PHONE")
    SERVICE_PHONE
  }

  public enum ServiceJumpType {
    @SerializedName("JUMP_MINI_PROGRAM")
    JUMP_MINI_PROGRAM,
    @SerializedName("JUMP_WEB_PAGE")
    JUMP_WEB_PAGE
  }

}
