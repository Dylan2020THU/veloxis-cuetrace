func QueryImages(config *wxpay_utility.MchConfig, request *QueryImagesRequest) (response io.ReadCloser, err error) {
	const (
		host   = "https://api.mch.weixin.qq.com"
		method = "GET"
		path   = "/v3/merchant-service/images/{media_id}"
	)

	reqUrl, err := url.Parse(fmt.Sprintf("%s%s", host, path))
	if err != nil {
		return nil, err
	}
	reqUrl.Path = strings.Replace(reqUrl.Path, "{media_id}", url.PathEscape(*request.MediaId), -1)
	httpRequest, err := http.NewRequest(method, reqUrl.String(), nil)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Accept", "*/*")
	httpRequest.Header.Set("Wechatpay-Serial", config.WechatPayPublicKeyId())
	authorization, err := wxpay_utility.BuildAuthorization(config.MchId(), config.CertificateSerialNo(), config.PrivateKey(), method, reqUrl.RequestURI(), nil)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Authorization", authorization)

	client := &http.Client{}
	httpResponse, err := client.Do(httpRequest)
	if err != nil {
		return nil, err
	}
	if httpResponse.StatusCode >= 200 && httpResponse.StatusCode < 300 {
		if httpResponse.Body == nil {
			return nil, fmt.Errorf("response body is nil")
		}
		// 注意：调用方在使用完返回的 io.ReadCloser 后必须调用 Close() 方法
		return httpResponse.Body, nil
	} else {
		defer httpResponse.Body.Close()
		respBody, err := wxpay_utility.ExtractResponseBody(httpResponse)
		if err != nil {
			return nil, err
		}
		return nil, wxpay_utility.NewApiException(
			httpResponse.StatusCode,
			httpResponse.Header,
			respBody,
		)
	}
}

type QueryImagesRequest struct {
	MediaId *string `json:"media_id,omitempty"`
}

func (o *QueryImagesRequest) MarshalJSON() ([]byte, error) {
	type Alias QueryImagesRequest

	}{

		MediaId: nil,
		Alias:   (*Alias)(o),
	}

}
