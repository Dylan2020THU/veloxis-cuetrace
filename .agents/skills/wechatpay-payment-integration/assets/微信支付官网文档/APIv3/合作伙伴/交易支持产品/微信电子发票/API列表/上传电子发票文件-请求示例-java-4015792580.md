## Java


需配合微信支付工具库 WXPayUtility 使用，请参考 [Java](https://pay.weixin.qq.com/doc/v3/partner/4014985777.md)

package com.java.demo;

import com.java.utils.WXPayUtility; // 引用微信支付工具库，参考：https://pay.weixin.qq.com/doc/v3/partner/4014985777

import com.google.gson.annotations.SerializedName;
import com.google.gson.annotations.Expose;
import okhttp3.MediaType;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

import java.io.IOException;
import okhttp3.MultipartBody;
import java.io.UncheckedIOException;
import java.security.PrivateKey;
import java.security.PublicKey;
import java.util.ArrayList;
import java.util.HashMap;

/**
 * 上传电子发票文件
 */
public class UploadFapiaoFile {
  private static String HOST = "https://api.mch.weixin.qq.com";
  private static String METHOD = "POST";
  private static String PATH = "/v3/new-tax-control-fapiao/fapiao-applications/upload-fapiao-file";

  public static void main(String[] args) {
    // TODO: 请准备商户开发必要参数，参考：https://pay.weixin.qq.com/doc/v3/partner/4013080340
    UploadFapiaoFile client = new UploadFapiaoFile(
      "19xxxxxxxx",                    // 商户号，是由微信支付系统生成并分配给每个商户的唯一标识符，商户号获取方式参考 https://pay.weixin.qq.com/doc/v3/partner/4013080340
      "1DDE55AD98Exxxxxxxxxx",         // 商户API证书序列号，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4013058924
      "/path/to/apiclient_key.pem",     // 商户API证书私钥文件路径，本地文件路径
      "PUB_KEY_ID_xxxxxxxxxxxxx",      // 微信支付公钥ID，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4013038589
      "/path/to/wxp_pub.pem"           // 微信支付公钥文件路径，本地文件路径
    );

    UploadFapiaoFileRequest request = new UploadFapiaoFileRequest();
    request.file = "file_bytes".getBytes();
    request.meta = new FileMeta();
request.meta.subMchid = "1900000109";
request.meta.fileType = FapiaoFileType.PDF;
request.meta.digestAlogrithm = DigestType.SM3;
request.meta.digest = "addca90a6a290b9642dbaccffdf01e8c22aa348940b2e96754169ba08c19e5db";
    try {
      UploadFapiaoFileResponse response = client.run(request);
        // TODO: 请求成功，继续业务逻辑
        System.out.println(response);
    } catch (WXPayUtility.ApiException e) {
        // TODO: 请求失败，根据状态码执行不同的逻辑
        e.printStackTrace();
    }
  }

  public UploadFapiaoFileResponse run(UploadFapiaoFileRequest request) {
    String uri = PATH;
     // 构建multipart/form-data请求体
     String metaJson = WXPayUtility.toJson(request.meta);
     RequestBody metaPart = RequestBody.create(MediaType.parse("application/json"), metaJson);
     //  todo: you_file_name替换为meta里面的文件名字段
     String contentType = WXPayUtility.getContentTypeByFileName("your_file_name");
     RequestBody filePart = RequestBody.create(MediaType.parse(contentType), request.file);
     MultipartBody requestBody = new MultipartBody.Builder()
             .setType(MultipartBody.FORM)
             .addFormDataPart("meta", null, metaPart)
             .addFormDataPart("file", "your_file_name", filePart)
             .build();

    Request.Builder reqBuilder = new Request.Builder().url(HOST + uri);
    reqBuilder.addHeader("Accept", "application/json");
    reqBuilder.addHeader("Wechatpay-Serial", wechatPayPublicKeyId);
    reqBuilder.addHeader("Authorization", WXPayUtility.buildAuthorization(mchid, certificateSerialNo, privateKey, METHOD, uri, metaJson));
    reqBuilder.method(METHOD, requestBody);
    Request httpRequest = reqBuilder.build();

    // 发送HTTP请求
    OkHttpClient client = new OkHttpClient.Builder().build();
    try (Response httpResponse = client.newCall(httpRequest).execute()) {
      String respBody = WXPayUtility.extractBody(httpResponse);
      if (httpResponse.code() >= 200 && httpResponse.code() < 300) {
        // 2XX 成功，验证应答签名
        WXPayUtility.validateResponse(this.wechatPayPublicKeyId, this.wechatPayPublicKey,
            httpResponse.headers(), respBody);

        // 从HTTP应答报文构建返回数据
        return WXPayUtility.fromJson(respBody, UploadFapiaoFileResponse.class);
      } else {
        throw new WXPayUtility.ApiException(httpResponse.code(), respBody, httpResponse.headers());
      }
    } catch (IOException e) {
      throw new UncheckedIOException("Sending request to " + uri + " failed.", e);
    }
  }

  private final String mchid;
  private final String certificateSerialNo;
  private final PrivateKey privateKey;
  private final String wechatPayPublicKeyId;
  private final PublicKey wechatPayPublicKey;

  public UploadFapiaoFile(String mchid, String certificateSerialNo, String privateKeyFilePath, String wechatPayPublicKeyId, String wechatPayPublicKeyFilePath) {
    this.mchid = mchid;
    this.certificateSerialNo = certificateSerialNo;
    this.privateKey = WXPayUtility.loadPrivateKeyFromPath(privateKeyFilePath);
    this.wechatPayPublicKeyId = wechatPayPublicKeyId;
    this.wechatPayPublicKey = WXPayUtility.loadPublicKeyFromPath(wechatPayPublicKeyFilePath);
  }

  public static class UploadFapiaoFileRequest {
    @SerializedName("file")
    public byte[] file;

    @SerializedName("meta")
    public FileMeta meta;
  }

  public static class UploadFapiaoFileResponse {
    @SerializedName("fapiao_media_id")
    public String fapiaoMediaId;
  }

  public static class FileMeta {
    @SerializedName("sub_mchid")
    public String subMchid;

    @SerializedName("file_type")
    public FapiaoFileType fileType;

    @SerializedName("digest_alogrithm")
    public DigestType digestAlogrithm;

    @SerializedName("digest")
    public String digest;
  }

  public enum FapiaoFileType {
    @SerializedName("PDF")
    PDF
  }

  public enum DigestType {
    @SerializedName("SM3")
    SM3
  }

}
