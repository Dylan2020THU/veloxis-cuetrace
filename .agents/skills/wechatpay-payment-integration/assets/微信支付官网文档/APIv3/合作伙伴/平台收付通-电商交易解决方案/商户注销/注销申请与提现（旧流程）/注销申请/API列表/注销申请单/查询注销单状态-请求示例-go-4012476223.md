func GetCancelApplication(config *wxpay_utility.MchConfig, request *GetCancelApplicationRequest) (response *CancelApplicationsEntity, err error) {
	const (
		host   = "https://api.mch.weixin.qq.com"
		method = "GET"
		path   = "/v3/ecommerce/account/cancel-applications/out-apply-no/{out_apply_no}"
	)

	reqUrl, err := url.Parse(fmt.Sprintf("%s%s", host, path))
	if err != nil {
		return nil, err
	}
	reqUrl.Path = strings.Replace(reqUrl.Path, "{out_apply_no}", url.PathEscape(*request.OutApplyNo), -1)
	httpRequest, err := http.NewRequest(method, reqUrl.String(), nil)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Accept", "application/json")
	httpRequest.Header.Set("Wechatpay-Serial", config.WechatPayPublicKeyId())
	authorization, err := wxpay_utility.BuildAuthorization(config.MchId(), config.CertificateSerialNo(), config.PrivateKey(), method, reqUrl.RequestURI(), nil)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Authorization", authorization)

	client := &http.Client{}
	httpResponse, err := client.Do(httpRequest)
	if err != nil {
		return nil, err
	}
	respBody, err := wxpay_utility.ExtractResponseBody(httpResponse)
	if err != nil {
		return nil, err
	}
	if httpResponse.StatusCode >= 200 && httpResponse.StatusCode < 300 {
		// 2XX 成功，验证应答签名
		err = wxpay_utility.ValidateResponse(
			config.WechatPayPublicKeyId(),
			config.WechatPayPublicKey(),
			&httpResponse.Header,
			respBody,
		)
		if err != nil {
			return nil, err
		}
		response := &CancelApplicationsEntity{}
		if err := json.Unmarshal(respBody, response); err != nil {
			return nil, err
		}

		return response, nil
	} else {
		return nil, wxpay_utility.NewApiException(
			httpResponse.StatusCode,
			httpResponse.Header,
			respBody,
		)
	}
}

type GetCancelApplicationRequest struct {
	OutApplyNo *string `json:"out_apply_no,omitempty"`
}

	type Alias GetCancelApplicationRequest
	a := &struct {

	}{
		// 序列化时移除非 Body 字段

type CancelApplicationsEntity struct {
	OutApplyNo   *string      `json:"out_apply_no,omitempty"`
	SubMchid     *string      `json:"sub_mchid,omitempty"`
	RejectReason *string      `json:"reject_reason,omitempty"`
	CancelState  *CancelState `json:"cancel_state,omitempty"`
	UpdateTime   *string      `json:"update_time,omitempty"`
}

type CancelState string

func (e CancelState) Ptr() *CancelState {
	return &e
}

const (
	CANCELSTATE_REVIEWING      CancelState = "REVIEWING"
	CANCELSTATE_REJECTED       CancelState = "REJECTED"
	CANCELSTATE_CANCEL_SUCCESS CancelState = "CANCEL_SUCCESS"
)
