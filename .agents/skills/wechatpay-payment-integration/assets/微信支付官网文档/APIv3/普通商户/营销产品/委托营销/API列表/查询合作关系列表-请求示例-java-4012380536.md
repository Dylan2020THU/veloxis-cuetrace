## Java


需配合微信支付工具库 WXPayUtility 使用，请参考 [Java](https://pay.weixin.qq.com/doc/v3/merchant/4014931831.md)

```
package com.java.demo;

import com.java.utils.WXPayUtility; // 引用微信支付工具库，参考：https://pay.weixin.qq.com/doc/v3/merchant/4014931831

import com.google.gson.annotations.SerializedName;
import com.google.gson.annotations.Expose;
import okhttp3.MediaType;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

import java.io.IOException;
import java.io.UncheckedIOException;
import java.security.PrivateKey;
import java.security.PublicKey;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * 查询合作关系列表
 */
public class ListPartnerships {
  private static String HOST = "https://api.mch.weixin.qq.com";
  private static String METHOD = "GET";
  private static String PATH = "/v3/marketing/partnerships";

  public static void main(String[] args) {
    // TODO: 请准备商户开发必要参数，参考：https://pay.weixin.qq.com/doc/v3/merchant/4013070756
    ListPartnerships client = new ListPartnerships(
      "19xxxxxxxx",                    // 商户号，是由微信支付系统生成并分配给每个商户的唯一标识符，商户号获取方式参考 https://pay.weixin.qq.com/doc/v3/merchant/4013070756
      "1DDE55AD98Exxxxxxxxxx",         // 商户API证书序列号，如何获取请参考 https://pay.weixin.qq.com/doc/v3/merchant/4013053053
      "/path/to/apiclient_key.pem",     // 商户API证书私钥文件路径，本地文件路径
      "PUB_KEY_ID_xxxxxxxxxxxxx",      // 微信支付公钥ID，如何获取请参考 https://pay.weixin.qq.com/doc/v3/merchant/4013038816
      "/path/to/wxp_pub.pem"           // 微信支付公钥文件路径，本地文件路径
    );

    ListPartnershipsRequest request = new ListPartnershipsRequest();
    request.limit = 1L;
    request.offset = 1L;
    request.partner = new PartnerInfo();
request.partner.type = PartnerType.APPID;
request.partner.appid = "wx4e1916a585d1f4e9";
request.partner.merchantId = "2480029552";
    request.authorizedData = new AuthorizedDataFilter();
request.authorizedData.businessType = AuthBusinessType.FAVOR_STOCK;
request.authorizedData.stockId = "2433405";
    try {
      ListPartnershipsResponse response = client.run(request);
        // TODO: 请求成功，继续业务逻辑
        System.out.println(response);
    } catch (WXPayUtility.ApiException e) {
        // TODO: 请求失败，根据状态码执行不同的逻辑
        e.printStackTrace();
    }
  }

  public ListPartnershipsResponse run(ListPartnershipsRequest request) {
    String uri = PATH;
    Map<String, Object> args = new HashMap<>();
    args.put("limit", request.limit);
    args.put("offset", request.offset);
    args.put("partner", request.partner);
    args.put("authorized_data", request.authorizedData);
    String queryString = WXPayUtility.urlEncode(args);
    if (!queryString.isEmpty()) {
        uri = uri + "?" + queryString;
    }

    Request.Builder reqBuilder = new Request.Builder().url(HOST + uri);
    reqBuilder.addHeader("Accept", "application/json");
    reqBuilder.addHeader("Wechatpay-Serial", wechatPayPublicKeyId);
    reqBuilder.addHeader("Authorization", WXPayUtility.buildAuthorization(mchid, certificateSerialNo, privateKey, METHOD, uri, null));
    reqBuilder.method(METHOD, null);
    Request httpRequest = reqBuilder.build();

    // 发送HTTP请求
    OkHttpClient client = new OkHttpClient.Builder().build();
    try (Response httpResponse = client.newCall(httpRequest).execute()) {
      String respBody = WXPayUtility.extractBody(httpResponse);
      if (httpResponse.code() >= 200 && httpResponse.code() < 300) {
        // 2XX 成功，验证应答签名
        WXPayUtility.validateResponse(this.wechatPayPublicKeyId, this.wechatPayPublicKey,
            httpResponse.headers(), respBody);

        // 从HTTP应答报文构建返回数据
        return WXPayUtility.fromJson(respBody, ListPartnershipsResponse.class);
      } else {
        throw new WXPayUtility.ApiException(httpResponse.code(), respBody, httpResponse.headers());
      }
    } catch (IOException e) {
      throw new UncheckedIOException("Sending request to " + uri + " failed.", e);
    }
  }

  private final String mchid;
  private final String certificateSerialNo;
  private final PrivateKey privateKey;
  private final String wechatPayPublicKeyId;
  private final PublicKey wechatPayPublicKey;

  public ListPartnerships(String mchid, String certificateSerialNo, String privateKeyFilePath, String wechatPayPublicKeyId, String wechatPayPublicKeyFilePath) {
    this.mchid = mchid;
    this.certificateSerialNo = certificateSerialNo;
    this.privateKey = WXPayUtility.loadPrivateKeyFromPath(privateKeyFilePath);
    this.wechatPayPublicKeyId = wechatPayPublicKeyId;
    this.wechatPayPublicKey = WXPayUtility.loadPublicKeyFromPath(wechatPayPublicKeyFilePath);
  }

  public static class ListPartnershipsRequest {
    @SerializedName("limit")
    @Expose(serialize = false)
    public Long limit;

    @SerializedName("offset")
    @Expose(serialize = false)
    public Long offset;

    @SerializedName("partner")
    @Expose(serialize = false)
    public PartnerInfo partner;

    @SerializedName("authorized_data")
    @Expose(serialize = false)
    public AuthorizedDataFilter authorizedData;
  }

  public static class ListPartnershipsResponse {
    @SerializedName("data")
    public List<PartnershipsEntity> data;

    @SerializedName("offset")
    public Long offset;

    @SerializedName("limit")
    public Long limit;

    @SerializedName("total_count")
    public Long totalCount;
  }

  public static class PartnerInfo {
    @SerializedName("type")
    public PartnerType type;

    @SerializedName("appid")
    public String appid;

    @SerializedName("merchant_id")
    public String merchantId;
  }

  public static class AuthorizedDataFilter {
    @SerializedName("business_type")
    public AuthBusinessType businessType;

    @SerializedName("stock_id")
    public String stockId;
  }

  public static class PartnershipsEntity {
    @SerializedName("partner")
    public PartnerInfo partner;

    @SerializedName("authorized_data")
    public AuthorizedData authorizedData;

    @SerializedName("build_time")
    public String buildTime;

    @SerializedName("terminate_time")
    public String terminateTime;

    @SerializedName("create_time")
    public String createTime;

    @SerializedName("update_time")
    public String updateTime;
  }

  public enum PartnerType {
    @SerializedName("APPID")
    APPID,
    @SerializedName("MERCHANT")
    MERCHANT
  }

  public enum AuthBusinessType {
    @SerializedName("FAVOR_STOCK")
    FAVOR_STOCK,
    @SerializedName("BUSIFAVOR_STOCK")
    BUSIFAVOR_STOCK
  }

  public static class AuthorizedData {
    @SerializedName("business_type")
    public AuthBusinessType businessType;

    @SerializedName("stock_id")
    public String stockId;
  }

}
```
