## Java


需配合微信支付工具库 WXPayUtility 使用，请参考 [Java](https://pay.weixin.qq.com/doc/brand/4015826861.md)

```
package com.java.demo;

import com.java.utils.WXPayBrandUtility; // 引用微信支付工具库，参考：https://pay.weixin.qq.com/doc/brand/4015826861

import com.google.gson.annotations.SerializedName;
import com.google.gson.annotations.Expose;
import okhttp3.MediaType;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

import java.io.IOException;
import java.io.UncheckedIOException;
import java.security.PrivateKey;
import java.security.PublicKey;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * 查询用户会员卡信息
 */
public class GetUserCard {
  private static String HOST = "https://api.mch.weixin.qq.com";
  private static String METHOD = "GET";
  private static String PATH = "/brand/card-member/user-cards/{user_card_code}";

  public static void main(String[] args) {
    // TODO: 请准备商户开发必要参数，参考：https://pay.weixin.qq.com/doc/brand/4015415289
    GetUserCard client = new GetUserCard(
      "xxxxxxxx",                    // 品牌ID，是由微信支付系统生成并分配给每个品牌方的唯一标识符，品牌ID获取方式参考 https://pay.weixin.qq.com/doc/brand/4015415289
      "1DDE55AD98Exxxxxxxxxx",         // 品牌API证书序列号，如何获取请参考 https://pay.weixin.qq.com/doc/brand/4015407570
      "/path/to/apiclient_key.pem",     // 品牌API证书私钥文件路径，本地文件路径
      "PUB_KEY_ID_xxxxxxxxxxxxx",      // 微信支付公钥ID，如何获取请参考 https://pay.weixin.qq.com/doc/brand/4015453439
      "/path/to/wxp_pub.pem"           // 微信支付公钥文件路径，本地文件路径
    );

    GetUserCardRequest request = new GetUserCardRequest();
    request.userCardCode = "478515832665";
    request.cardId = "pbLatjvWOibDc5-TBnbUk1pD12o0";
    request.openid = "obLatjnx9gnqzS4myYGmLZ7LgLBA";
    try {
      UserCard response = client.run(request);
        // TODO: 请求成功，继续业务逻辑
        System.out.println(response);
    } catch (WXPayBrandUtility.ApiException e) {
        // TODO: 请求失败，根据状态码执行不同的逻辑
        e.printStackTrace();
    }
  }

  public UserCard run(GetUserCardRequest request) {
    String uri = PATH;
    uri = uri.replace("{user_card_code}", WXPayBrandUtility.urlEncode(request.userCardCode));
    Map<String, Object> args = new HashMap<>();
    args.put("card_id", request.cardId);
    args.put("openid", request.openid);
    String queryString = WXPayBrandUtility.urlEncode(args);
    if (!queryString.isEmpty()) {
        uri = uri + "?" + queryString;
    }

    Request.Builder reqBuilder = new Request.Builder().url(HOST + uri);
    reqBuilder.addHeader("Accept", "application/json");
    reqBuilder.addHeader("Wechatpay-Serial", wechatPayPublicKeyId);
    reqBuilder.addHeader("Authorization", WXPayBrandUtility.buildAuthorization(brand_id, certificateSerialNo, privateKey, METHOD, uri, null));
    reqBuilder.method(METHOD, null);
    Request httpRequest = reqBuilder.build();

    // 发送HTTP请求
    OkHttpClient client = new OkHttpClient.Builder().build();
    try (Response httpResponse = client.newCall(httpRequest).execute()) {
      String respBody = WXPayBrandUtility.extractBody(httpResponse);
      if (httpResponse.code() >= 200 && httpResponse.code() < 300) {
        // 2XX 成功，验证应答签名
        WXPayBrandUtility.validateResponse(this.wechatPayPublicKeyId, this.wechatPayPublicKey,
            httpResponse.headers(), respBody);

        // 从HTTP应答报文构建返回数据
        return WXPayBrandUtility.fromJson(respBody, UserCard.class);
      } else {
        throw new WXPayBrandUtility.ApiException(httpResponse.code(), respBody, httpResponse.headers());
      }
    } catch (IOException e) {
      throw new UncheckedIOException("Sending request to " + uri + " failed.", e);
    }
  }

  private final String brand_id;
  private final String certificateSerialNo;
  private final PrivateKey privateKey;
  private final String wechatPayPublicKeyId;
  private final PublicKey wechatPayPublicKey;

  public GetUserCard(String brand_id, String certificateSerialNo, String privateKeyFilePath, String wechatPayPublicKeyId, String wechatPayPublicKeyFilePath) {
    this.brand_id = brand_id;
    this.certificateSerialNo = certificateSerialNo;
    this.privateKey = WXPayBrandUtility.loadPrivateKeyFromPath(privateKeyFilePath);
    this.wechatPayPublicKeyId = wechatPayPublicKeyId;
    this.wechatPayPublicKey = WXPayBrandUtility.loadPublicKeyFromPath(wechatPayPublicKeyFilePath);
  }

  public static class GetUserCardRequest {
    @SerializedName("card_id")

    public String cardId;

    @SerializedName("user_card_code")
    @Expose(serialize = false)
    public String userCardCode;

    @SerializedName("openid")
    @Expose(serialize = false)
    public String openid;
  }

  public static class UserCard {

    @SerializedName("card_id")

    @SerializedName("openid")

    @SerializedName("card_type")

    @SerializedName("phone_number")
    public String phoneNumber;

    @SerializedName("level")
    public String level;

    @SerializedName("valid_date_information")
    public DateInfo validDateInformation;

    @SerializedName("pickup_time")
    public String pickupTime;

    @SerializedName("user_information")
    public UserInfo userInformation;

    @SerializedName("attach")
    public String attach;

    @SerializedName("user_card_state")
    public UserCardState userCardState;

    @SerializedName("invalid_reason")
    public String invalidReason;

    @SerializedName("invalid_time")
    public String invalidTime;

    @SerializedName("create_time")
    public String createTime;

    @SerializedName("modify_time")
    public String modifyTime;
  }

  public enum CardType {
    @SerializedName("PURCHASE")
    PURCHASE,
    @SerializedName("NORMAL")
    NORMAL,
    @SerializedName("BALANCE")
    BALANCE
  }

  public static class DateInfo {
    @SerializedName("type")
    public DateType type;

    @SerializedName("available_begin_time")
    public String availableBeginTime;

    @SerializedName("available_end_time")
    public String availableEndTime;

    @SerializedName("available_day_after_receive")
    public Long availableDayAfterReceive;
  }

  public static class UserInfo {
    @SerializedName("common_field_list")
    public List<UserInfoCommonField> commonFieldList;

    @SerializedName("custom_field_list")
    public List<UserInfoCustomField> customFieldList;
  }

  public enum UserCardState {
    @SerializedName("UNACTIVATED")
    UNACTIVATED,
    @SerializedName("EFFECTIVE")
    EFFECTIVE,
    @SerializedName("EXPIRED")
    EXPIRED,
    @SerializedName("INVALID")
    INVALID
  }

  public enum DateType {
    @SerializedName("FIX_TIME_RANGE")
    FIX_TIME_RANGE,
    @SerializedName("FIX_TERM")
    FIX_TERM,
    @SerializedName("PERMANENT")
    PERMANENT
  }

  public static class UserInfoCommonField {
    @SerializedName("name")
    public CommonFieldFlag name;

    @SerializedName("value")
    public String value;
  }

  public static class UserInfoCustomField {
    @SerializedName("name")
    public String name;

    @SerializedName("user_chosen_values")
    public List<String> userChosenValues;
  }

  public enum CommonFieldFlag {
    @SerializedName("USER_FORM_FLAG_SEX")
    USER_FORM_FLAG_SEX,
    @SerializedName("USER_FORM_FLAG_NAME")
    USER_FORM_FLAG_NAME,
    @SerializedName("USER_FORM_FLAG_BIRTHDAY")
    USER_FORM_FLAG_BIRTHDAY,
    @SerializedName("USER_FORM_FLAG_ADDRESS")
    USER_FORM_FLAG_ADDRESS,
    @SerializedName("USER_FORM_FLAG_EMAIL")
    USER_FORM_FLAG_EMAIL,
    @SerializedName("USER_FORM_FLAG_CITY")
    USER_FORM_FLAG_CITY
  }

}
```
