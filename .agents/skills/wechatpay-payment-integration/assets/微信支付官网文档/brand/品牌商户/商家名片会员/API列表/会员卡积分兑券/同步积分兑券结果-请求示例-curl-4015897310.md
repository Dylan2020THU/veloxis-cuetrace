## curl


> POST

```
curl -X POST \
  https://api.mch.weixin.qq.com/brand/card-member/user-points/exchange-coupon/confirm \
  -H "Authorization: WECHATPAY-BRAND-SHA256-RSA2048 brand_id=\"XXXX\",..." \
  -H "Accept: application/json" \
  -H "Wechatpay-Serial: PUB_KEY_ID_XXXX"  \
  -H "Content-Type: application/json" \
  -d '{
    "record_id" : "pbLatjvWOibDc5-TBnbUk1pD12o0",
    "exchange_coupon_template_id" : "0Flpo7FqTQ",
    "card_id" : "pbLatjvWOibDc5-TBnbUk1pD12o0",
    "openid" : "obLatjnx9gnqzS4myYGmLZ7LgLBA",
    "user_card_code" : "478515832665",
    "result" : "POINT_EXCHANGE_COUPON_ALLOW",
    "reject_reason" : "积分不足"
  }'

```
