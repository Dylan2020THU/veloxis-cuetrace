## curl


> PATCH

```
curl -X PATCH \
  https://api.mch.weixin.qq.com/brand/marketing/delivery-plan/delivery-plans/12000 \
  -H "Authorization: WECHATPAY-BRAND-SHA256-RSA2048 brand_id=\"XXXX\",..." \
  -H "Accept: application/json" \
  -H "Wechatpay-Serial: PUB_KEY_ID_XXXX"  \
  -H "Content-Type: application/json" \
  -d '{
    "modify_content" : {
      "plan_name" : "冬季促销",
      "delivery_end_time" : "2025-01-01T00:00:00+08:00",
      "total_count" : 1,
      "user_limit" : 1,
      "daily_limit" : 1,
      "recommend_word" : "冬季优惠"
    },
    "out_request_no" : "abcd-1234-1000"
  }'

```
