func GetImageGenerationTask(config *wxpay_brand_utility.BrandConfig, request *GetImageGenerationTaskRequest) (response *ImageGenerationTask, err error) {
	const (
		host   = "https://api.mch.weixin.qq.com"
		method = "GET"
		path   = "/brand/marketing/product-coupon/image-generation-tasks/{task_id}"
	)

	reqUrl, err := url.Parse(fmt.Sprintf("%s%s", host, path))
	if err != nil {
		return nil, err
	}
	reqUrl.Path = strings.Replace(reqUrl.Path, "{task_id}", url.PathEscape(*request.TaskId), -1)
	query := reqUrl.Query()
	if request.ImageGenerationType != nil {
		query.Add("image_generation_type", fmt.Sprintf("%v", *request.ImageGenerationType))
	}
	reqUrl.RawQuery = query.Encode()
	httpRequest, err := http.NewRequest(method, reqUrl.String(), nil)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Accept", "application/json")
	httpRequest.Header.Set("Wechatpay-Serial", config.WechatPayPublicKeyId())
	authorization, err := wxpay_brand_utility.BuildAuthorization(config.BrandId(), config.CertificateSerialNo(), config.PrivateKey(), method, reqUrl.RequestURI(), nil)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Authorization", authorization)

	client := &http.Client{}
	httpResponse, err := client.Do(httpRequest)
	if err != nil {
		return nil, err
	}
	respBody, err := wxpay_brand_utility.ExtractResponseBody(httpResponse)
	if err != nil {
		return nil, err
	}
	if httpResponse.StatusCode >= 200 && httpResponse.StatusCode < 300 {
		// 2XX 成功，验证应答签名
		err = wxpay_brand_utility.ValidateResponse(
			config.WechatPayPublicKeyId(),
			config.WechatPayPublicKey(),
			&httpResponse.Header,
			respBody,
		)
		if err != nil {
			return nil, err
		}
		response := &ImageGenerationTask{}
		if err := json.Unmarshal(respBody, response); err != nil {
			return nil, err
		}

		return response, nil
	} else {
		return nil, wxpay_brand_utility.NewApiException(
			httpResponse.StatusCode,
			httpResponse.Header,
			respBody,
		)
	}
}

type GetImageGenerationTaskRequest struct {
	TaskId              *string              `json:"task_id,omitempty"`
	ImageGenerationType *ImageGenerationType `json:"image_generation_type,omitempty"`
}

func (o *GetImageGenerationTaskRequest) MarshalJSON() ([]byte, error) {
	type Alias GetImageGenerationTaskRequest
	a := &struct {
		TaskId              *string              `json:"task_id,omitempty"`
		ImageGenerationType *ImageGenerationType `json:"image_generation_type,omitempty"`
		*Alias
	}{
		// 序列化时移除非 Body 字段
		TaskId:              nil,
		ImageGenerationType: nil,
		Alias:               (*Alias)(o),
	}
	return json.Marshal(a)
}

type ImageGenerationTask struct {
	TaskId              *string              `json:"task_id,omitempty"`
	ImageGenerationType *ImageGenerationType `json:"image_generation_type,omitempty"`
	TaskState           *TaskState           `json:"task_state,omitempty"`
	CombineImageResult  *CombineImageResult  `json:"combine_image_result,omitempty"`
	CutOutResult        *CutOutResult        `json:"cut_out_result,omitempty"`
}

type ImageGenerationType string

func (e ImageGenerationType) Ptr() *ImageGenerationType {
	return &e
}

const (
	IMAGEGENERATIONTYPE_COMBINE_IMAGE ImageGenerationType = "COMBINE_IMAGE"
	IMAGEGENERATIONTYPE_CUT_OUT       ImageGenerationType = "CUT_OUT"
)

type TaskState string

func (e TaskState) Ptr() *TaskState {
	return &e
}

const (
	TASKSTATE_PROCESSING TaskState = "PROCESSING"
	TASKSTATE_FAIL       TaskState = "FAIL"
	TASKSTATE_SUCCESS    TaskState = "SUCCESS"
)

type CombineImageResult struct {
	ImageUrl *string `json:"image_url,omitempty"`
}

type CutOutResult struct {
	ImageUrl *string `json:"image_url,omitempty"`
}
