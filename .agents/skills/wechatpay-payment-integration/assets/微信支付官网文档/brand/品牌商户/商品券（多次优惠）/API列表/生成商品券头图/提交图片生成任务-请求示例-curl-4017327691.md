## curl


> POST

```
curl -X POST \
  https://api.mch.weixin.qq.com/brand/marketing/product-coupon/image-generation-tasks \
  -H "Authorization: WECHATPAY-BRAND-SHA256-RSA2048 brand_id=\"XXXX\",..." \
  -H "Accept: application/json" \
  -H "Wechatpay-Serial: PUB_KEY_ID_XXXX"  \
  -H "Content-Type: application/json" \
  -d '{
    "task_id" : "image_generation_task_1",
    "image_generation_type" : "COMBINE_IMAGE",
    "combine_image" : {
      "scope" : "ALL",
      "type" : "NORMAL",
      "usage_mode" : "SINGLE",
      "normal_coupon" : {
        "threshold" : 10000,
        "discount_amount" : 100
      },
      "discount_coupon" : {
        "threshold" : 10000,
        "percent_off" : 30
      },
      "exchange_coupon" : {
        "threshold" : 10000,
        "exchange_price" : 100
      },
      "background_color" : "#ff5733"
    },
    "cut_out" : {
      "image_url" : "https://wxpaylogo.qpic.cn/wxpaylogo/example.jpg"
    }
  }'

```
