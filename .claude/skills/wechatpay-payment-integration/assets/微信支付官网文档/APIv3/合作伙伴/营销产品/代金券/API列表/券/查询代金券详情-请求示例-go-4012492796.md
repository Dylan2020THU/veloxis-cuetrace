func QueryCoupon(config *wxpay_utility.MchConfig, request *QueryCouponRequest) (response *Coupon, err error) {
	const (
		host   = "https://api.mch.weixin.qq.com"
		method = "GET"
		path   = "/v3/marketing/favor/users/{openid}/coupons/{coupon_id}"
	)

	reqUrl, err := url.Parse(fmt.Sprintf("%s%s", host, path))
	if err != nil {
		return nil, err
	}
	reqUrl.Path = strings.Replace(reqUrl.Path, "{coupon_id}", url.PathEscape(*request.CouponId), -1)
	reqUrl.Path = strings.Replace(reqUrl.Path, "{openid}", url.PathEscape(*request.Openid), -1)
	query := reqUrl.Query()
	if request.Appid != nil {
		query.Add("appid", *request.Appid)
	}
	reqUrl.RawQuery = query.Encode()
	httpRequest, err := http.NewRequest(method, reqUrl.String(), nil)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Accept", "application/json")
	httpRequest.Header.Set("Wechatpay-Serial", config.WechatPayPublicKeyId())
	authorization, err := wxpay_utility.BuildAuthorization(config.MchId(), config.CertificateSerialNo(), config.PrivateKey(), method, reqUrl.RequestURI(), nil)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Authorization", authorization)

	client := &http.Client{}
	httpResponse, err := client.Do(httpRequest)
	if err != nil {
		return nil, err
	}
	respBody, err := wxpay_utility.ExtractResponseBody(httpResponse)
	if err != nil {
		return nil, err
	}
	if httpResponse.StatusCode >= 200 && httpResponse.StatusCode < 300 {
		// 2XX 成功，验证应答签名
		err = wxpay_utility.ValidateResponse(
			config.WechatPayPublicKeyId(),
			config.WechatPayPublicKey(),
			&httpResponse.Header,
			respBody,
		)
		if err != nil {
			return nil, err
		}
		response := &Coupon{}
		if err := json.Unmarshal(respBody, response); err != nil {
			return nil, err
		}

		return response, nil
	} else {
		return nil, wxpay_utility.NewApiException(
			httpResponse.StatusCode,
			httpResponse.Header,
			respBody,
		)
	}
}

type Coupon struct {
	StockCreatorMchid       *string             `json:"stock_creator_mchid,omitempty"`
	StockId                 *string             `json:"stock_id,omitempty"`
	CouponId                *string             `json:"coupon_id,omitempty"`
	CutToMessage            *CutTypeMsg         `json:"cut_to_message,omitempty"`
	CouponName              *string             `json:"coupon_name,omitempty"`
	Status                  *string             `json:"status,omitempty"`
	Description             *string             `json:"description,omitempty"`
	CreateTime              *string             `json:"create_time,omitempty"`
	CouponType              *string             `json:"coupon_type,omitempty"`
	NoCash                  *bool               `json:"no_cash,omitempty"`
	AvailableBeginTime      *string             `json:"available_begin_time,omitempty"`
	AvailableEndTime        *string             `json:"available_end_time,omitempty"`
	Singleitem              *bool               `json:"singleitem,omitempty"`
	NormalCouponInformation *FixedValueStockMsg `json:"normal_coupon_information,omitempty"`
	OutRequestNo            *string             `json:"out_request_no,omitempty"`

	BusinessType            *BusinessType       `json:"business_type,omitempty"`
	DiscountMsg             *DiscountMsg        `json:"discount_msg,omitempty"`
}

type CutTypeMsg struct {
	SinglePriceMax *int64 `json:"single_price_max,omitempty"`

}

type FixedValueStockMsg struct {
	CouponAmount       *int64 `json:"coupon_amount,omitempty"`
	TransactionMinimum *int64 `json:"transaction_minimum,omitempty"`
}

type BusinessType string

func (e BusinessType) Ptr() *BusinessType {
	return &e
}

const (

)

type DiscountMsg struct {
	DiscountAmountMax  *int64 `json:"discount_amount_max,omitempty"`
	DiscountPercent    *int64 `json:"discount_percent,omitempty"`
	TransactionMinimum *int64 `json:"transaction_minimum,omitempty"`
}
